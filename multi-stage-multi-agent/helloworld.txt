pipeline {
    agent none

    stages {

        stage('Back-end Build') {
            agent {
                docker {
                    image 'maven:3.8.1-adoptopenjdk-11'
                }
            }

            steps {
                sh '''
                javac HelloWorld.java
                java HelloWorld
                '''
            }
        }

        stage('Front-end Build') {
            agent {
                docker {
                    image 'node:16-alpine'
                }
            }

            steps {
                sh '''
                node app.js
                '''
            }
        }

    }
}
